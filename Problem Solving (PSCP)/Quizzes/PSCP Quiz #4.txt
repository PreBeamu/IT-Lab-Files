    if user_correct and not pass_corret:
        print("wrong password")
    elif not user_correct and pass_corret:
        print("wrong username")
    elif not user_correct and not pass_corret:
        print("wrong username and password")
    elif user_correct and pass_corret:
        print("login succeeded")
